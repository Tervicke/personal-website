  10. Write an Object Oriented Program in C++, to implement circle class to find area and circumference of a circle using functions void area(), void circum(). 
    
#include <iostream>
using namespace std;
class circle{
	int radius = 0;
	public:
		circle(int r){
			radius =r;
		}
		void area(){
			int area = 3.14*radius*radius; 
			cout << "Area is " << area << endl;
		}
		void circum(){
			int circ = 2*3.14*radius;
			cout << "circumference is " << circ;
		}
};
int main(){
	circle c1(5);
	c1.area();
	c1.circum();
	return 0;
}
  11. Implement class temperature to convert degree Fahrenheit (F) to degree Celsius (C). Using formulae C =( F -32/9)*5.

using namespace std;
class Temperature{
	public:
		int To_Fahrenheit(int C){
			return (C/5)+(32/9);
		}
		int TO_celsius(int F){
			 return ( F -32/9)*5;
		}
};
int main(){
	Temperature t;
	cout << t.TO_celsius(80);
	cout << t.To_Fahrenheit(105);
	return 0;
}
  12. Write a program in C++, to find the largest number in an array of 10 integers. 
    
#include <iostream>
using namespace std;
int main(){
	int ar[10] = {1 ,2 , 3 , 4 , 5 , 7 , 9 , 10 , 15 , 17};
	int max = 0;
	for(int i =0 ; i < 10 ; i++){
		if(max < ar[i]){
			max = ar[i];
		}
	}
	cout << "Max value in the array is " << max; // 17
	return 0;
}
  13. Write a program in C++, to calculate x^y , using the function void power (int, int); 
    
#include <iostream>
using namespace std;
void power(int x , int y ){
	int ans = 1;
	for(int i = 0 ; i < y ; i++){
		ans = ans * x;		
	}
	cout << ans << endl;
}
int main(){
	power(2,5); // 32 output 
	return 0;
}
