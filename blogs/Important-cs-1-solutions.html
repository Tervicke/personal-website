  3. Write a program in C++, to find factorial of a given number using for loop 
    					
#include <iostream>
using namespace std;
int main(){
	int number,ans=1;
	cout << "Input the number: " ;
	cin >> number;
	for(int i = 1 ; i <= number;i++){
		ans = i * ans;
	}
	cout << "The factorial of the number is " << ans << endl;
	return 0;
}

  5. Write a program in C++, to print first 15 terms of Fibonacci series

	cout << 0 << endl << 1 << endl;

		cout << ans << endl;
